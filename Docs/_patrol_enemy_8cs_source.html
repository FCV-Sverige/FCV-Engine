93 private void PatrolMovement()
94 {
95 if (patrolPoints.Count == 0)
96 return;
97
98 // Calculate the interpolation factor based on the speed
99 Vector3 target = CurrentTarget;
100 float travelTime = Vector3.Distance(transform.position, target) / patrolSpeed;
101 float lerpFactor = Time.deltaTime / travelTime;
102
103 if (Vector3.Distance(rb.position, target) > stopDistance)
104 {
105 Vector3 end = target;
106 transform.position = Vector3.Lerp(transform.position, end, lerpFactor);
107 }
108 else
109 {
110 // Reached the patrol point
111 if (movingForward)
112 {
113 currentPointIndex++;
114 if (currentPointIndex >= patrolPoints.Count)
115 {
116 movingForward = false;
117 }
118 }
119 else
120 {
121 currentPointIndex--;
122 if (currentPointIndex < 0)
123 {
124 movingForward = true;
125 }
126 }
127 }
