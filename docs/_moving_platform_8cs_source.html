109 private int GetNextIndex(int currentIndex, int arrayLength, LoopType movementType)
110 {
111 switch (movementType)
112 {
113 // Loop behavior: Move to the next index, wrapping around to 0 if needed
114 case LoopType.LOOP:
115 return (currentIndex + 1) % arrayLength;
116
117 // Ping-Pong-behavior: Move to the next index, reversing direction at the ends
118 case LoopType.PINGPONG when !pingPongReversing:
119 {
120 if (currentIndex + 1 < arrayLength) return currentIndex + 1;
121 // Reverse direction at the end
122 pingPongReversing = true;
123 return currentIndex - 1;
124 }
125
126 // Ping-Pong-Behaviour: Control reversing in ping-pong behaviour
127 case LoopType.PINGPONG when currentIndex - 1 < 0:
128 // Reverse direction at the start
129 pingPongReversing = false;
130 return currentIndex + 1;
131
132 case LoopType.PINGPONG:
133 return currentIndex - 1;
134
135 default:
136 return currentIndex; // Default case (should not be reached)
137 }
